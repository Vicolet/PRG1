insert, replace, erase
Qu’imprime le programme suivant, en supposant avoir supprimé les éventuelles lignes fautives.

 #include <iostream>

 using namespace std;

 int main() {
    string s1 = "ABC";
    string s2 = "123";
    s1.insert(1, s2);
    cout << '|' << s1 << '|' << endl;

    s1 = "ABC";
    s2 = "123";
    // s1.insert(4, s2);
    cout << '|' << s1 << '|' << endl;

    s1 = "ABC";
    s2 = "123";
    s1.insert(2, s2, 1, 2);
    cout << '|' << s1 << '|' << endl;

    s1 = "ABC";
    s2 = "123";
    s1.insert(2, s2, 0, string::npos);
    cout << '|' << s1 << '|' << endl;

    s1 = "ABC";
    s1.insert(3, "123", 2);
    cout << '|' << s1 << '|' << endl;

    s1 = "ABC";
    s1.insert(1, 2, '3');
    cout << '|' << s1 << '|' << endl;

    s1 = "ABCDEF";
    s2 = "123";
    s1.replace(1, 2, s2);
    cout << '|' << s1 << '|' << endl;

    s1 = "ABCDEF";
    s2 = "123";
    s1.replace(2, 4, s2);
    cout << '|' << s1 << '|' << endl;

    s1 = "ABCDEF";
    s2 = "123";
    s1.replace(1, 2, s2, 1, 2);
    cout << '|' << s1 << '|' << endl;

    s1 = "ABCDEF";
    s1.replace(2, 3, "123", 2);
    cout << '|' << s1 << '|' << endl;

    s1 = "ABCDEF";
    s1.erase();
    cout << '|' << s1 << '|' << endl;

    s1 = "ABCDEF";
    s1.erase(2);
    cout << '|' << s1 << '|' << endl;

    s1 = "ABCDEF";
    s1.erase(2, 2);
    cout << '|' << s1 << '|' << endl;
 }
